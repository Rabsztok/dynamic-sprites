def initialize(command, path, options = {})
  @interface = Interface.new
  @command = command
  @path = Pathname.new(path) unless path.nil?
  @path ||= File.join(sass_directory, 'dynamic-sprites.sass') unless command.nil?
  @options = options
  @options[:filename] ||= "#{@path}.png"
end
